JVM Status Checker

When deploying from Staging to Production, we track the status of our rolling JVM restart with a page that looks like 90% gobbledegook to the average user. I took the initiative to create this little JS tool which, when run in the browser console, parses out just the relevant information and displays it for optimal legibility.


// get web content
var txt = $('body').outerHTML;
// remove html tags
txt = txt.replace(/<.*?>/g,'').trim();

// regex pattern to match current JVM
var currentJvmRgx = /JVM:\s.+?[app|int]00\d?/;
var currentJvmTxt = txt.match(currentJvmRgx);
var currentJvmParser = /JVM:\s.+(app|int)(00\d)/;
var currentJvmDisplay = currentJvmParser.exec(currentJvmTxt);
var jvmDisplay = "Current JVM = " + currentJvmDisplay[1] + currentJvmDisplay[2] + " *\n\n";

// regex pattern to match each JVM
var jvmTxtRgx = 
    /JVM\s.+[app|int]00\d[\S\s]+?Restart:\s[A-z]{3}\s[A-z]{3}\s\d+\s\d{2}:\d{2}:\d{2}\sP[DS]T\s\d{4}\sUptime:\s.+/g;

// put JVM text into array jvmText
var jvmText = txt.match(jvmTxtRgx);
var jvmInfo = [];
// regex pattern parsing data into matching groups
var regexParser = 
    /(app|int)00(\d)[\S\s]+?([A-z]{3})\s([A-z]{3})\s(\d+)\s(\d{2}:\d{2}:\d{2})\sP[DS]T\s(\d{4})\sUptime:\s(.+)/;

// put JVM data into array jvmInfo
for (var i = 0; i < jvmText.length; i++) {	
	jvmInfo[i] = regexParser.exec(jvmText[i]);
}

var jvmObjects = [];
var j = 1;
// put each JVM into array jvmObjects, with named data members, in order
for (i = 0; i < jvmInfo.length; i++) {
	// create JVM object
	var jvm = {name:jvmInfo[i][1] + "00" + jvmInfo[i][2], type:jvmInfo[i][1], number:jvmInfo[i][2], weekday:jvmInfo[i][3], 
               month:jvmInfo[i][4], date:jvmInfo[i][5], time:jvmInfo[i][6], year:jvmInfo[i][7], uptime:jvmInfo[i][8]};

	// place admin JVM last in array
	if (jvm.type == "int") {
		jvmObjects.push(jvm);
	// place first JVM at start of array
	} else if (jvmObjects[0] == null) {

		jvmObjects.unshift(jvm);
	// check if this JVM belongs first
	} else if (jvmObjects[0].number > jvm.number || jvmObjects[0].type == "int") {
		jvmObjects.unshift(jvm)
	} else {
		for (j = 1; j <= jvmObjects.length; j++) {
			if (jvmObjects[j] == null || jvmObjects[j].number > jvm.number || jvmObjects[j].type == "int") {
				jvmObjects.splice(j, 0, jvm);
				break;
			}
		}
	}
}
for (i = 0; i < jvmObjects.length; i++) {
	if (currentJvmDisplay[1] + currentJvmDisplay[2] == jvmObjects[i].name) {
		jvmObjects[i].name += "*";
	}
	jvmDisplay += ("JVM " + jvmObjects[i].name + " last restarted at " + jvmObjects[i].time + 
                   " on " + jvmObjects[i].weekday + ", " + jvmObjects[i].month + " " + jvmObjects[i].date + "\n"
                   + "Refreshed " + jvmObjects[i].uptime + " ago\n\n");
}

A scheduled batch file moves new XML files into a workspace folder, runs the API, and archives the files, processing thousands of emails per hour. The revamped emails were estimated to have generated over $48K of additional revenue in the first month.

    //* recursively collect all files within "file"
    public void addTree(File file, Collection emailFiles) {
    	if (file.isFile() && !file.isHidden() && file.getName().indexOf(".xml") > -1 ) {
			emailFiles.add(file);
		}
    	File[] children = file.listFiles();
    	if ( children != null ) {
    		for (File child : children) {
    			if ( !"Archive".equalsIgnoreCase(child.getName())) {
	    			if (child.isFile() && !child.isHidden() && child.getName().indexOf(".xml") > -1 ) {
	    				emailFiles.add(child);
	    			}
    			}
    		}
    	}
    	debug("Files in queue:");
    	for (File f : emailFiles) {
    		debug("(file): " + f.getAbsolutePath());
    	}
    } // end of addTree()

			return contact;

    //* turns XML from CW into key/value pairs in a LinkedHashMap
    public MutableBoolean parseXML (LinkedHashMap thisMessage, File file, /* String filePath,*/ LinkedHashMap carriers, 
    			LinkedHashMap nonInvItems, LinkedHashMap utilItems) {
    	MutableBoolean sendIndicator = new MutableBoolean(true);
    	List addlOrderMessages = new ArrayList();
    	log("Parsing file");
    	String certonaItems = "";
        try{
        	JAXBContext jContext = JAXBContext.newInstance(Message.class);
            Unmarshaller unmarshallerObj = jContext.createUnmarshaller();
            Message message = (Message) unmarshallerObj.unmarshal(file);
            
            if (message != null) {
            	putMessageData(thisMessage, message);
            	
            	if (message.getEmail() != null) {
            		Email email = message.getEmail();
            		putEmailData(thisMessage, email);
            
            		if (email.getCustomerSoldTo() != null) {
            			CustomerSoldTo customerSoldTo = email.getCustomerSoldTo();
            			putCustomerSoldToData(thisMessage, customerSoldTo);
            		
            			if (customerSoldTo.getCustomerSoldToHistory() != null) {
            				CustomerSoldToHistory customerSoldToHistory = customerSoldTo.getCustomerSoldToHistory();
            				putCustomerSoldToHistory(thisMessage, customerSoldToHistory);
            			}
            			
            			if (email.getCustomerShipTo() != null) {
            				CustomerShipTo customerShipTo = email.getCustomerShipTo();
            				putCustomerShipTo(thisMessage, customerShipTo);
            			}
            			
            			if (email.getOrder() != null) {
            				Order order = email.getOrder();
            				putOrder(thisMessage, order);
            
            				if (order.getOrderDetails() != null) {          
				            	List orderDetails = order.getOrderDetails().getOrderDetail();
				                if (orderDetails != null)
				                	certonaItems = putOrderDetails(thisMessage, orderDetails, certonaItems);
				            }
            
				            if (order.getOrderPayments() != null) {
				            	OrderPayments orderPayments = order.getOrderPayments();
				            	if (orderPayments.getOrderPaymentCount() != null)
				            		thisMessage.put("order_payment_count", orderPayments.getOrderPaymentCount());
				            	List orderPaymentList = orderPayments.getOrderPayment();
				            	if (orderPaymentList != null)
				            		putOrderPayments(thisMessage, orderPaymentList);
				            }
            
				            if (order.getInvoice() != null) {
				            	Invoice invoice = order.getInvoice();
				            	putInvoice(thisMessage, invoice, carriers);
				            	List invoiceDetails = invoice.getInvoiceDetails().getInvoiceDetail();
				            	if (invoiceDetails != null) {
					            	certonaItems = putInvoiceDetails(thisMessage, invoiceDetails, certonaItems, 
					            				nonInvItems, utilItems, addlOrderMessages, sendIndicator);
				            	}
				            	if (invoice.getTrackingNumbers() != null) {
				            		List trackingNumbers = invoice.getTrackingNumbers().getTrackingNumber();
				            		putTrackingNumbers(thisMessage, trackingNumbers, carriers);
				            	}
				            		
				            }
				            
				            if (order.getOrderMessages() != null || addlOrderMessages.size() > 0) {
				            	List orderMessages = null;
				            	if (order.getOrderMessages() != null ) 
				            		orderMessages = order.getOrderMessages().getOrderMessage();
				            	putOrderMessages(thisMessage, orderMessages, addlOrderMessages);
				            }
            			}
            
			            if (email.getEmailTemplateText().getBeforeBodyTextLines() != null) {
			            	List beforeBodyTextLines = email.getEmailTemplateText().getBeforeBodyTextLines().getBeforeBodyTextLine();
			            	putBeforeBodyTextLines(thisMessage, beforeBodyTextLines);
			            }
			            
			            if (email.getEmailTemplateText().getAfterBodyTextLines() != null) {
			            	List afterBodyTextLines = email.getEmailTemplateText().getAfterBodyTextLines().getAfterBodyTextLine();
			            	putAfterBodyTextLines(thisMessage, afterBodyTextLines);
			            }
            		}
            	}
            }
            
            if (certonaItems != "")
            	thisMessage.put("cw_certona_items", certonaItems);
           
            printAllData(thisMessage);
            
        }catch(Exception e){
            e.printStackTrace();
            sendIndicator.setValue(false);
        }
        return sendIndicator;
    } // end of parseXML()
    
    //* format an amount of pennies into currency
    public static String parseCurrency(String val) {
    	if (val != null) {
	    	try{
	    		DecimalFormat formatter = new DecimalFormat("$#,##0.00;-$#,##0.00");
	    		return formatter.format(Double.parseDouble(val) / 100);
	    	} catch (NumberFormatException e) {}
    	}
    	return null;
    } // end of parseCurrency()
    
    //* format a CW style date into a more standard one
    public static String parseDate(String dateIn) {
    	if (dateIn != null) {
	    	try{        	
	        	DateFormat dfIn = new SimpleDateFormat("yyyy-MM-dd");
	        	DateFormat dfOut = new SimpleDateFormat("M/d/yyyy");
	    		Date dateOut = dfIn.parse(dateIn);
	    		return dfOut.format(dateOut);
	    	} catch (Exception e) {}
    	}
    	return null;
    } // end of parseDate()
    
    //* format a 24-hour time into an am/pm one
    public static String parseTime (String timeIn) {
    	if (timeIn != null) {
	    	try{
	    		SimpleDateFormat tfIn = new SimpleDateFormat("HH:mm:ss");
	    		SimpleDateFormat tfOut = new SimpleDateFormat("h:mm:ss a");
	    		Date timeOut = tfIn.parse(timeIn);
	    		return tfOut.format(timeOut);
	    	} catch (Exception e) {}
    	}
    	return null;
    } // end of parseTime()

Products Missing From Employee Site

Our employee discount is based on products being placed in certain directories, so after each web launch, we need to make sure all relevant items are in the appropriate places. This simple Oracle SQL query finds all current products that are not in any of the employee categories.


Alter session set current_schema = title9stg;

select distinct p.code, psd.on_sale, p.DATE_CREATED, sum(i.stock)
    from product p inner join
        product_category_link pcl on pcl.PRODUCT_ID = p.PRODUCT_ID inner join
        category c on c.CATEGORY_ID = pcl.CATEGORY_ID inner join
        product_sku_link psl on psl.PRODUCT_ID = p.PRODUCT_ID inner join
        site_product_link spl on spl.product_id = p.product_id and spl.site_id = 1 inner join
        inventory i on i.SKU_ID = psl.SKU_ID inner join
        product_site_data psd on psd.PRODUCT_ID = p.PRODUCT_ID and psd.SITE_ID = 1
    where p.status = 1 and p.DELETE_FLAG = 0 and
        regexp_like (p.code, '^\d{6}$') and
        p.code not in (select distinct p.code
            from product p inner join
            product_category_link pcl on pcl.PRODUCT_ID = p.PRODUCT_ID inner join
            category c on c.CATEGORY_ID = pcl.CATEGORY_ID
            where c.code in ('EMP_BRAS', 'EMP_TOPS', 'EMP_BOTTOMS',
            'EMP_DRESSES', 'EMP_SWIM', 'EMP_SHOES', 'EMP_SALE', 'BLACKOUT')
        )
    group by p.code, psd.on_sale, p.DATE_CREATED
    order by psd.on_sale desc, sum(i.stock) desc;
